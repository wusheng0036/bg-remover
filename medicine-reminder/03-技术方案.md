## 🏗️ 技术选型

### 推荐方案：微信云开发

**理由**:
- ✅ 免运维，专注业务逻辑
- ✅ 免费额度够 MVP 使用（月活 10 万以下免费）
- ✅ 原生集成微信能力（登录、支付、消息）
- ✅ 快速迭代，一人可全栈

### 技术栈

| 层级 | 技术 | 说明 |
|------|------|------|
| 前端 | 原生微信小程序 | 无需框架，最直接 |

## 🔌 云函数设计

### 1. login - 用户登录

```javascript
// 云函数：login/index.js
exports.main = async (event, context) => {
  const { OPENID } = cloud.getWXContext();
  
  // 查询或创建用户
  const db = cloud.database();
  const user = await db.collection('users').where({ _openid: OPENID }).get();
  
  if (user.data.length === 0) {
    // 新用户
    await db.collection('users').add({
      data: {
        _openid: OPENID,
        nickname: event.userInfo?.nickName || '用户',
        avatar: event.userInfo?.avatarUrl || '',
        createdAt: Date.now(),
        lastLoginAt: Date.now()
      }
    });
  } else {
    // 更新最后登录时间
    await db.collection('users').doc(user.data[0]._id).update({
      data: { lastLoginAt: Date.now() }
    });
  }
  
  return { openid: OPENID };
};
```

### 2. addMedicine - 添加药品

```javascript
// 云函数：addMedicine/index.js
exports.main = async (event, context) => {
  const { OPENID } = cloud.getWXContext();
  const { medicine } = event;
  
  const db = cloud.database();
  
  // 验证数据
  if (!medicine.name || !medicine.times || medicine.times.length === 0) {
    return { success: false, error: '药品名称和服药时间必填' };
  }
  
  // 创建药品记录
  const result = await db.collection('medicines').add({
    data: {
      _openid: OPENID,
      ...medicine,
      enabled: true,
      createdAt: Date.now(),
      updatedAt: Date.now()
    }
  });
  
  // 创建提醒任务
  await createReminderTasks(OPENID, result._id, medicine);
  
  return { success: true, id: result._id };
};

// 辅助函数：创建提醒任务
async function createReminderTasks(openid, medicineId, medicine) {
  const db = cloud.database();
  const tasks = [];
  
  // 为未来 7 天创建提醒任务
  for (let i = 0; i < 7; i++) {
    const date = new Date();
    date.setDate(date.getDate() + i);
    const dateStr = formatDate(date);
    
    for (const time of medicine.times) {
      const scheduledTime = `${dateStr} ${time}`;
      const sendTime = new Date(scheduledTime);
      sendTime.setMinutes(sendTime.getMinutes() - (medicine.reminderAdvance || 15));
      
      tasks.push({
        _openid: openid,
        medicineId,
        scheduledTime,
        sendTime: sendTime.toISOString(),
        status: 'pending',
        createdAt: Date.now()
      });
    }
  }
  
  if (tasks.length > 0) {
    await db.collection('reminder_tasks').add({ data: tasks });
  }
}
```

### 3. sendReminder - 发送提醒（定时触发）

```javascript
// 云函数：sendReminder/index.js
const cloud = require('wx-server-sdk');
cloud.init({ env: cloud.DYNAMIC_CURRENT_ENV });

exports.main = async (event, context) => {
  const db = cloud.database();
  const now = new Date();
  
  // 查询待发送的提醒
  const tasks = await db.collection('reminder_tasks')
    .where({
      status: 'pending',
      sendTime: db.command.lte(now.toISOString())
    })
    .get();
  
  for (const task of tasks.data) {
    try {
      // 获取用户信息
      const user = await db.collection('users')
        .where({ _openid: task._openid })
        .get();
      
      if (user.data.length === 0) continue;
      
      // 获取药品信息
      const medicine = await db.collection('medicines')
        .where({ _id: task.medicineId })
        .get();
      
      if (medicine.data.length === 0) continue;
      
      // 发送订阅消息
      await cloud.openapi.subscribeMessage.send({
        touser: task._openid,
        templateId: 'YOUR_TEMPLATE_ID',
        data: {
          thing1: { value: medicine.data[0].name },
          thing2: { value: medicine.data[0].instruction },
          time3: { value: task.scheduledTime }
        }
      });
      
      // 更新任务状态
      await db.collection('reminder_tasks').doc(task._id).update({
        data: { status: 'sent', sentAt: Date.now() }
      });
      
    } catch (err) {
      console.error('发送提醒失败:', err);
      await db.collection('reminder_tasks').doc(task._id).update({
        data: { status: 'failed', error: err.message }
      });
    }
  }
  
  return { success: true, sent: tasks.data.length };
};
```

### 4. checkIn - 服药打卡

```javascript
// 云函数：checkIn/index.js
exports.main = async (event, context) => {
  const { OPENID } = cloud.getWXContext();
  const { medicineId, scheduledTime, photo } = event;
  
  const db = cloud.database();
  const now = new Date();
  
  // 创建服药记录
  const record = await db.collection('records').add({
    data: {
      _openid: OPENID,
      medicineId,
      scheduledTime,
      actualTime: now.toISOString(),
      status: 'completed',
      photo: photo || '',
      isMakeup: false,
      createdAt: Date.now()
    }
  });
  
  // 检查是否漏服超时，通知家属
  const medicine = await db.collection('medicines').doc(medicineId).get();
  const scheduled = new Date(scheduledTime);
  const diffMinutes = (now - scheduled) / 60000;
  
  if (diffMinutes > 30) {
    await notifyFamilyMembers(OPENID, medicine.data, 'late');
  }
  
  return { success: true, id: record._id };
};

// 辅助函数：通知家属
async function notifyFamilyMembers(patientOpenid, medicine, type) {
  const db = cloud.database();
  
  // 查询家属
  const families = await db.collection('families')
    .where({
      patientOpenid,
      status: 'active',
      notifyOnMissed: true
    })
    .get();
  
  for (const family of families.data) {
    try {
      await cloud.openapi.subscribeMessage.send({
        touser: family.caregiverOpenid,
        templateId: 'FAMILY_NOTIFY_TEMPLATE_ID',
        data: {
          thing1: { value: family.patientName },
          thing2: { value: medicine.name },
          thing3: { value: type === 'missed' ? '漏服' : '延迟服药' }
        }
      });
    } catch (err) {
      console.error('通知家属失败:', err);
    }
  }
}
```

## ⚠️ 注意事项

1. **订阅消息需要用户主动触发** - 不能首次进入就弹窗，要在用户添加药品后引导授权
2. **医疗类目资质** - MVP 定位为"健康工具"，避免"医疗建议"相关表述
3. **数据隐私** - 健康数据敏感，数据库权限设置为"仅创建者可读写"
4. **消息频率限制** - 微信对订阅消息有频率限制，避免骚扰用户
